<div class="section active" id="sec-grundlagen">

  <h2 class="section-heading">Grundlagen des SCM</h2>

  <p class="section-sub">Definitionen, Zielsetzungen und das Kernparadigma.</p>


  <div class="card">

    <div class="card-title"><div class="icon">📖</div>Definition nach Buscher (1999)</div>

    <div class="definition-box">

      SCM ist ein strategisches Unternehmensführungskonzept, das die Geschäftsprozesse entlang der Versorgungskette — vom ersten Rohstofflieferanten bis zum Endverbraucher — zur Kundenzufriedenheit gestaltet.

    </div>

  </div>


  <div class="card">

    <div class="card-title"><div class="icon">🔗</div>Definition nach Vahrenkamp (1998)</div>

    <div class="definition-box">

      Die ganzheitliche Betrachtung der Logistikkette zielt auf die Abstimmung der Güterströme im gesamten Netz ab.

    </div>

    <p style="font-size:0.875rem;color:var(--text-muted);line-height:1.6;">Schönsleben/Hieber (2004) ergänzen: SCM umfasst das Management <strong>innerhalb und außerhalb der Unternehmensgrenzen</strong> über den gesamten Produktlebenszyklus.</p>

  </div>


  <div class="card">

    <div class="card-title"><div class="icon">🎯</div>Zielsetzungen des SCM</div>

    <ul class="sum-list">

      <li><div class="dot"></div><div>Orientierung am Endkundennutzen & Steigerung der Kundenzufriedenheit</div></li>

      <li><div class="dot"></div><div>Reduktion der Bestände (→ Kostensenkung, Kapitalbindung reduzieren)</div></li>

      <li><div class="dot"></div><div>Verstetigung des Güterstroms → einfachere Steuerung</div></li>

      <li><div class="dot"></div><div>Rasche und effiziente Anpassung an Marktänderungen</div></li>

      <li><div class="dot"></div><div>Verkürzung der Auftragsdurchlaufzeiten</div></li>

      <li><div class="dot"></div><div>Vermeidung von Out-of-Stock (OOS)-Situationen</div></li>

    </ul>

    <div class="memory-box">

      <div class="memory-box-label">Merkhilfe — "KRAVE"</div>

      <strong>K</strong>apital binden reduzieren · <strong>R</strong>asche Anpassung · <strong>A</strong>uftragsdurchlaufzeiten kürzen · <strong>V</strong>ermeiden von OOS · <strong>E</strong>ndkunde zufriedenstellen

    </div>

  </div>


  <div class="card">

    <div class="card-title"><div class="icon">⚡</div>Wesentliches Paradigma</div>

function drawLosgroesse() {

  const canvas = document.getElementById('losgroesse-canvas');

  if (!canvas) return;

  const ctx = canvas.getContext('2d');

  const W = canvas.offsetWidth || 500;

  canvas.width = W;

  canvas.height = 220;

  const rk = parseInt(document.getElementById('rustkostenSlider').value);

  document.getElementById('rustkostenVal').textContent = rk.toLocaleString('de') + ' €';


  const lkRate = 0.4;

  const points = [];

  for (let q = 500; q <= 12000; q += 200) {

    const fix = rk / q;

    const vari = (q / 2) * lkRate / 1000;

    points.push({q, fix, vari, total: fix + vari});

  }

  const maxCost = Math.max(...points.map(p => p.total));

  const pad = {l:50, r:20, t:20, b:30};

  const cw = W - pad.l - pad.r;

  const ch = canvas.height - pad.t - pad.b;


  function xt(q) { return pad.l + (q - 500) / 11500 * cw; }

  function yt(c) { return pad.t + ch - (c / (maxCost * 1.1)) * ch; }


  ctx.clearRect(0, 0, W, canvas.height);

  ctx.strokeStyle = '#E2E0D8'; ctx.lineWidth = 1;

  for (let i = 0; i <= 4; i++) {

    const y = pad.t + i * ch / 4;

    ctx.beginPath(); ctx.moveTo(pad.l, y); ctx.lineTo(W - pad.r, y); ctx.stroke();

  }


  function drawLine(data, color, key) {

    ctx.beginPath(); ctx.strokeStyle = color; ctx.lineWidth = 2;

    data.forEach((p, i) => { i === 0 ? ctx.moveTo(xt(p.q), yt(p[key])) : ctx.lineTo(xt(p.q), yt(p[key])); });

    ctx.stroke();

  }

  drawLine(points, '#1D9E75', 'fix');

  drawLine(points, '#378ADD', 'vari');

  drawLine(points, '#EF9F27', 'total');


  const minPt = points.reduce((a, b) => a.total < b.total ? a : b);

  ctx.setLineDash([4,3]); ctx.strokeStyle = '#EF9F27'; ctx.lineWidth = 1.5;

  ctx.beginPath(); ctx.moveTo(xt(minPt.q), pad.t); ctx.lineTo(xt(minPt.q), pad.t + ch); ctx.stroke();

  ctx.setLineDash([]);

  ctx.fillStyle = '#EF9F27'; ctx.font = '11px DM Sans, sans-serif';

  ctx.fillText('Optimum: ' + minPt.q.toLocaleString('de'), xt(minPt.q) + 4, pad.t + 14);


  ctx.fillStyle = '#6b6b6b'; ctx.font = '11px DM Sans, sans-serif';

  ctx.fillText('0', pad.l - 10, pad.t + ch);

  ctx.fillText('Losgröße', W/2 - 20, canvas.height - 2);

}

let qIdx = 0, score = 0, answered = false;


function renderQuestion() {

  if (qIdx >= questions.length) { showScore(); return; }

  const q = questions[qIdx];

  answered = false;

  document.getElementById('quiz-counter').textContent = 'Frage ' + (qIdx+1) + ' von ' + questions.length;

  document.getElementById('quiz-q').textContent = q.q;

  document.getElementById('quiz-feedback').classList.remove('show','ok','fail');

  document.getElementById('quiz-next-btn').style.display = 'none';

  const optsEl = document.getElementById('quiz-opts');

  optsEl.innerHTML = '';

  q.opts.forEach((opt, i) => {

    const btn = document.createElement('button');

    btn.className = 'quiz-option';

    btn.textContent = opt;

    btn.onclick = () => selectAnswer(i);

    optsEl.appendChild(btn);

  });

}


function selectAnswer(i) {

  if (answered) return;

  answered = true;

  const q = questions[qIdx];

  const btns = document.querySelectorAll('.quiz-option');

  btns.forEach((b, idx) => {

    b.classList.add('disabled');

    if (idx === q.correct) b.classList.add('correct');

    else if (idx === i && i !== q.correct) b.classList.add('wrong');

  });

  const fb = document.getElementById('quiz-feedback');

  if (i === q.correct) {

    score++;

    fb.textContent = '✓ Richtig! ' + q.exp;

    fb.className = 'quiz-feedback show ok';

  } else {

    fb.textContent = '✗ Leider falsch. ' + q.exp;

    fb.className = 'quiz-feedback show fail';

  }

  document.getElementById('quiz-next-btn').style.display = 'inline-block';

}


function nextQuestion() {

  qIdx++;

  renderQuestion();

}


function showScore() {

  document.getElementById('quiz-counter').textContent = '';

  document.getElementById('quiz-q').textContent = '';

  document.getElementById('quiz-opts').innerHTML = '';

  document.getElementById('quiz-feedback').classList.remove('show');

  document.getElementById('quiz-next-btn').style.display = 'none';

  const pct = Math.round(score / questions.length * 100);

  const emoji = pct >= 80 ? '🎉' : pct >= 60 ? '👍' : '📚';

  const msg = pct >= 80 ? 'Ausgezeichnet! Prüfungsfit!' : pct >= 60 ? 'Gut! Noch ein wenig üben.' : 'Noch üben — du schaffst das!';

  document.getElementById('quiz-score-area').style.display = 'block';

  document.getElementById('quiz-score-area').innerHTML = `

    <div class="quiz-score-box">

      <div style="font-size:2rem;margin-bottom:0.5rem;">${emoji}</div>

      <div class="quiz-score-big">${score} / ${questions.length}</div>

      <div class="quiz-score-label">${pct}% – ${msg}</div>

      <button class="quiz-btn" style="margin-top:1rem;" onclick="restartQuiz()">Quiz wiederholen</button>

    </div>`;

}


function restartQuiz() {

  qIdx = 0; score = 0;

  document.getElementById('quiz-score-area').style.display = 'none';

  renderQuestion();

}
